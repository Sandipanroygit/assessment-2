"use client";

import Link from "next/link";
import { useEffect, useState } from "react";
import { supabase } from "@/lib/supabaseClient";
import { products } from "@/data/products";

type Profile = { full_name?: string; role?: string };

const stats = [
  { label: "Active modules", value: "120", delta: "+8 this month" },
  { label: "Products live", value: "48", delta: "4 featured" },
  { label: "Orders this week", value: "86", delta: "12 pending" },
  { label: "Revenue (Gé¦)", value: "7.2L", delta: "Gåæ 18% vs last month" },
];

const curriculumActions = [
  "Upload videos, Python files, docs",
  "Tag by grade, subject, module",
  "Publish/unpublish instantly",
];

const productActions = [
  "Add drone and hands-on learning SKUs",
  "Attach images, price, delivery ETA",
  "Feature products in ads",
  "Edit listings and delete retired SKUs",
];

const orderActions = ["Track status", "View receipts", "Export reports"];

const analytics = [
  { title: "Engagement", metric: "6.4k views", detail: "Avg 3.2 min per module" },
  { title: "Conversions", metric: "12.7%", detail: "CTA to checkout" },
  { title: "Promo performance", metric: "Gé¦2.1L", detail: "Ad-driven revenue" },
];

const formatPrice = (value: number) =>
  new Intl.NumberFormat("en-IN", { style: "currency", currency: "INR", maximumFractionDigits: 0 }).format(
    value,
  );

export default function AdminPage() {
  const [profile, setProfile] = useState<Profile | null>(null);

  useEffect(() => {
    const loadProfile = async () => {
      const { data } = await supabase.auth.getUser();
      const user = data.user;
      if (!user) return;
      const { data: profileData } = await supabase
        .from("profiles")
        .select("full_name, role")
        .eq("id", user.id)
        .maybeSingle();
      const fallbackName = profileData?.full_name ?? user.user_metadata?.full_name ?? "Admin";
      setProfile(profileData ? { ...profileData, full_name: fallbackName } : { full_name: fallbackName, role: "admin" });
    };
    loadProfile();
  }, []);

  return (
    <main className="section-padding space-y-8">
      <div className="flex items-center justify-between flex-wrap gap-4">
        <div>
          <p className="text-accent-strong uppercase text-xs tracking-[0.2em]">Admin</p>
          <h1 className="text-3xl font-semibold text-white">
            Hi {profile?.full_name ?? "Admin"}, here&apos;s your control room
          </h1>
          <p className="text-slate-300 text-sm mt-2">
            Manage curriculum, products, orders, and promotions in one dashboard.
          </p>
        </div>
        <div className="flex gap-3">
          <Link
            href="/"
            className="px-4 py-2 rounded-xl border border-white/10 text-sm text-white hover:border-accent-strong"
          >
            Back to Home
          </Link>
          <button
            onClick={() => supabase.auth.signOut()}
            className="px-4 py-2 rounded-xl bg-accent text-slate-900 font-semibold shadow-glow"
          >
            Sign out
          </button>
        </div>
      </div>

      <div className="grid sm:grid-cols-2 lg:grid-cols-4 gap-4">
        {stats.map((item) => (
          <div key={item.label} className="glass-panel rounded-2xl p-4 space-y-2">
            <p className="text-sm text-slate-400">{item.label}</p>
            <p className="text-2xl font-semibold text-white">{item.value}</p>
            <p className="text-xs text-accent-strong">{item.delta}</p>
          </div>
        ))}
      </div>

      <div className="grid md:grid-cols-2 gap-6">
        <div className="glass-panel rounded-2xl p-6 space-y-3">
          <div className="flex items-center justify-between">
            <h2 className="text-xl font-semibold text-white">Curriculum management</h2>
            <button className="text-sm px-3 py-1 rounded-lg bg-white/10 border border-white/15 text-white">
              Upload content
            </button>
          </div>
          <p className="text-slate-300 text-sm">
            Select grade, subject, and module to add videos, Python code files, and documents.
          </p>
          <ul className="list-disc list-inside text-sm text-slate-200 space-y-1">
            {curriculumActions.map((action) => (
              <li key={action}>{action}</li>
            ))}
          </ul>
        </div>

        <div className="glass-panel rounded-2xl p-6 space-y-3">
          <div className="flex items-center justify-between">
            <h2 className="text-xl font-semibold text-white">Product management</h2>
            <button className="text-sm px-3 py-1 rounded-lg bg-white/10 border border-white/15 text-white">
              Add product
            </button>
          </div>
          <p className="text-slate-300 text-sm">
            Add, edit, or delete drone kits and hands-on bundles with price and delivery details.
          </p>
          <ul className="list-disc list-inside text-sm text-slate-200 space-y-1">
            {productActions.map((action) => (
              <li key={action}>{action}</li>
            ))}
          </ul>
        </div>
      </div>

      <div className="glass-panel rounded-2xl p-6 space-y-4">
        <div className="flex items-center justify-between gap-3">
          <h2 className="text-lg font-semibold text-white">Product catalogue</h2>
          <Link href="/shop" className="text-sm text-accent-strong hover:underline">
            View shop
          </Link>
        </div>
        <p className="text-sm text-slate-300">
          List every shop item and trigger edit or delete actions directly from the control room.
        </p>
        <div className="overflow-auto">
          <table className="min-w-full text-sm text-slate-200">
            <thead>
              <tr className="text-left text-slate-400 border-b border-white/10">
                <th className="py-2 pr-3">Name</th>
                <th className="py-2 pr-3">Price</th>
                <th className="py-2 pr-3">Delivery</th>
                <th className="py-2 pr-3">Actions</th>
              </tr>
            </thead>
            <tbody>
              {products.map((product) => (
                <tr key={product.id} className="border-b border-white/5">
                  <td className="py-2 pr-3 font-semibold text-white">{product.name}</td>
                  <td className="py-2 pr-3">{formatPrice(product.price)}</td>
                  <td className="py-2 pr-3">{product.deliveryEta}</td>
                  <td className="py-2 pr-3">
                    <div className="flex gap-2">
                      <button
                        className="px-3 py-1 rounded-lg bg-white/10 border border-white/15 text-white text-xs"
                        onClick={() => alert(`Edit ${product.name}`)}
                      >
                        Edit
                      </button>
                      <button
                        className="px-3 py-1 rounded-lg border border-red-400/40 text-red-200 text-xs hover:bg-red-500/10 transition"
                        onClick={() => alert(`Delete ${product.name}`)}
                      >
                        Delete
                      </button>
                    </div>
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      </div>

      <div className="grid md:grid-cols-3 gap-6">
        <div className="glass-panel rounded-2xl p-6 space-y-3">
          <div className="flex items-center justify-between">
            <h2 className="text-lg font-semibold text-white">Orders</h2>
            <button className="text-sm px-3 py-1 rounded-lg bg-white/10 border border-white/15 text-white">
              View all
            </button>
          </div>
          <ul className="list-disc list-inside text-sm text-slate-200 space-y-1">
            {orderActions.map((action) => (
              <li key={action}>{action}</li>
            ))}
          </ul>
          <div className="rounded-xl border border-white/10 p-3 text-sm text-slate-300">
            Live status: 12 pending, 58 processing, 16 delivered.
          </div>
        </div>

        <div className="glass-panel rounded-2xl p-6 space-y-3">
          <div className="flex items-center justify-between">
            <h2 className="text-lg font-semibold text-white">Promotions</h2>
            <button className="text-sm px-3 py-1 rounded-lg bg-white/10 border border-white/15 text-white">
              Create promo
            </button>
          </div>
          <p className="text-slate-300 text-sm">
            Run ads like Gé¦9,999/year drone curriculum and 15% off kits. Track conversions instantly.
          </p>
          <div className="rounded-xl border border-accent/30 bg-accent/10 text-accent-strong p-3 text-sm">
            Featured: Free AR/VR module for new signups.
          </div>
        </div>

        <div className="glass-panel rounded-2xl p-6 space-y-3">
          <h2 className="text-lg font-semibold text-white">Analytics</h2>
          <p className="text-slate-300 text-sm">
            Sales reports, user engagement, and promotion performance updated in real time.
          </p>
          <div className="grid grid-cols-1 gap-3">
            {analytics.map((item) => (
              <div key={item.title} className="rounded-xl border border-white/10 p-3">
                <p className="text-slate-400 text-xs uppercase tracking-wide">{item.title}</p>
                <p className="text-white font-semibold text-lg">{item.metric}</p>
                <p className="text-slate-300 text-xs">{item.detail}</p>
              </div>
            ))}
          </div>
        </div>
      </div>
    </main>
  );
}
